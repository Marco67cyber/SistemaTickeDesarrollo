    t.id AS Ticket_ID,
    c.nombre AS Cliente_Afectado,
    u.nombre AS Ejecutivo_Responsable,
    t.descripcion_problema,
    t.fecha_creacion,
    t.estado_tique
FROM Tiques t
INNER JOIN Clientes c ON t.cliente_rut = c.rut
INNER JOIN Usuarios u ON t.ejecutivo_creador_id = u.id;
