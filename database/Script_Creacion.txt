-- 1. Crear la base de datos con el nombre correcto
CREATE DATABASE IF NOT EXISTS db_consultenos;

-- 2. Seleccionar la base de datos recién creada para trabajar en ella
USE db_consultenos;

-- 3. Tabla para gestionar los roles (Jefe de Mesa, Ejecutivo de Mesa, etc.)
CREATE TABLE Roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL
);

-- 4. Tabla de Usuarios (Cumple con HJM-01 y HJM-02)
CREATE TABLE Usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    rut VARCHAR(12) UNIQUE NOT NULL,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    rol_id INT,
    estado ENUM('activo', 'inactivo') DEFAULT 'activo', -- RN-02: No se eliminan, se desactivan
    FOREIGN KEY (rol_id) REFERENCES Roles(id)
);

-- 5. Tabla de Clientes (Base para HEM-01)
CREATE TABLE Clientes (
    rut VARCHAR(12) PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    telefono VARCHAR(15),
    email VARCHAR(100)
);

-- 6. Tabla principal de Tiques (Base para HEM-01 y HEM-02)
CREATE TABLE Tiques (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cliente_rut VARCHAR(12),
    ejecutivo_creador_id INT,
    descripcion_problema TEXT NOT NULL,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    estado_tique ENUM('abierto', 'en_revision', 'derivado', 'cerrado') DEFAULT 'abierto',
    FOREIGN KEY (cliente_rut) REFERENCES Clientes(rut),
    FOREIGN KEY (ejecutivo_creador_id) REFERENCES Usuarios(id)
);
